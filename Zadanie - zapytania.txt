SELECT DESCRIPTION, I.NAME 
FROM ISSUES
JOIN ISSUESLISTS I ON ISSUES.ISSUESLIST_ID = I.ID;

SELECT DESCRIPTION, U.ID 
FROM ISSUES
JOIN USERS U ON ISSUES.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM USERS U  
JOIN ISSUES ON U.ID = ISSUES.USER_ID_ASSIGNEDTO
GROUP BY U.ID;
